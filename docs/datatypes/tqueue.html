TQueue

TQueue

A convenience implementation of a queue in the STM monad, built on top of TVar.

import cats.effect.IO
import cats.effect.unsafe.implicits.global
import cats.syntax.semigroup._
import cats.instances.string._

import io.github.timwspence.cats.stm.STM

val stm = STM.runtime[IO].unsafeRunSync()
// stm: STM[IO] = io.github.timwspence.cats.stm.STM$$anon$1@2a0567c9
import stm._

val txn: Txn[String] = for {
  tqueue <- TQueue.empty[String]
  _      <- tqueue.put("hello")
  _      <- tqueue.put("world")
  hello  <- tqueue.read
  world  <- tqueue.peek
} yield hello |+| world
// txn: Txn[String] = Bind(
//   Bind(
//     Alloc(Delay(cats.effect.IO$$$Lambda$9868/0x0000000102d17c40@4c855059)),
//     scala.Function1$$Lambda$9900/0x0000000102db5840@40cc3ab7
//   ),
//   <function1>
// )

val result = stm.commit(txn).unsafeRunSync()
// result: String = "helloworld"
