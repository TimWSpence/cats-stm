TQueue

A convenience implementation of a queue in the STM monad, built on top of TVar.

import cats.effect.IO
// import cats.effect.IO

import cats.syntax.semigroup._
// import cats.syntax.semigroup._

import cats.instances.string._
// import cats.instances.string._

import io.github.timwspence.cats.stm.{STM, TQueue}
// import io.github.timwspence.cats.stm.{STM, TQueue}

val txn: STM[String] = for {
  tqueue <- TQueue.empty[String]
  _      <- tqueue.put("hello")
  _      <- tqueue.put("world")
  hello  <- tqueue.read
  world  <- tqueue.peek
} yield hello |+| world
// txn: io.github.timwspence.cats.stm.STM[String] = io.github.timwspence.cats.stm.STM@74f14c4e

val result = txn.commit[IO].unsafeRunSync
// result: String = helloworld
